// models/categories.js
import pool from './db.js';

async function getCategoriesByProjectId(projectId) {
  const query = `
    SELECT c.category_id, c.category_name
    FROM project_categories pc
    JOIN categories c ON pc.category_id = c.category_id
    WHERE pc.project_id = $1;
  `;
  const result = await pool.query(query, [projectId]);
  return result.rows;
}

export { getCategoriesByProjectId };

Next question (Topic 3 – Controllers):
Please paste a controller route handler that uses a route parameter (for example, a route like /project/:projectId or similar). After you share it, we'll discuss what the route parameter does and why you chose that approach.


// Controller for project details page
const showProjectDetailsPage = async (req, res) => {
  try {
    const projectId = req.params.id;   // <-- route parameter
    const project = await getProjectDetails(projectId);
    const categories = await getCategoriesByProjectId(projectId);

    if (!project) {
      return res.status(404).render('404', { message: 'Project not found' });
    }

    const title = 'Service Project Details';
    res.render('project', { title, project, categories });
  } catch (err) {
    console.error('Error loading project details:', err);
    res.status(500).render('500', { message: 'Server error' });
  }
};

This handler is tied to a route like /project/:id. The :id part is the route parameter, and Express makes it available as req.params.id. That’s how the controller knows which project to fetch from the database.

The reason for this approach is flexibility: instead of writing separate routes for each project, you can use one dynamic route handler that works for any project ID. It keeps your code DRY (Don’t Repeat Yourself) and makes the app scalable.

// Handle 404 - Page Not Found
app.use((req, res, next) => {
  res.status(404).render('404', { 
    title: '404 - Page Not Found',
    message: 'Sorry, the page you are looking for does not exist.'
  });
});

// Handle 500 - Server Error
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).render('500', { 
    title: '500 - Server Error',
    message: 'Something went wrong on our end.'
  });
});

This code ensures that if a user tries to visit a route that doesn’t exist, they’ll see a friendly 404 page. And if something goes wrong inside the server (like a database error), they’ll see a 500 page instead of a crash.

The reason for structuring it this way is to give users clear feedback when something fails, while also keeping your application secure (you don’t expose raw error details to the user).
