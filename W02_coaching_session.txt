// db/projects.js
import pool from '../db.js';

export async function getProjectsWithCategoriesAndOrg() {
  const query = `
    SELECT sp.project_name, o.organization_name, c.category_name
    FROM service_projects sp
    JOIN organizations o ON sp.organization_id = o.organization_id
    JOIN project_categories pc ON sp.project_id = pc.project_id
    JOIN categories c ON pc.category_id = c.category_id;
  `;
  const { rows } = await pool.query(query);
  return rows;
}

This function connects to the database using the pool object. It runs a SQL query that joins projects, organizations, and categories. The results (rows) are then returned to the application.

The data flow works like this:

The query runs in the database.

The database sends back the results.

The Node.js function collects those results and passes them to the Express route.

The route then sends the data to the EJS template for rendering on the client side.

I implemented it this way because it keeps the database logic separate from the route logic, which makes the code cleaner and easier to maintain.
